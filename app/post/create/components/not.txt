const handleCategoryChange = (category: string) => {
    // Update internal state
    if (internalSelectedCategories.includes(category)) {
      setInternalSelectedCategories(
        internalSelectedCategories.filter((c) => c !== category)
      );
    } else {
      setInternalSelectedCategories([...internalSelectedCategories, category]);
    }

    // Call the parent's onChange function with the updated selected categories
    onChange(internalSelectedCategories);
  };
